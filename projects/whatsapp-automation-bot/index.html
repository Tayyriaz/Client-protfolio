#!/usr/bin/env python3
"""
WhatsApp Automation Bot
A comprehensive business automation solution for WhatsApp
"""
from flask import Flask, request, jsonify
from flask_cors import CORS
import openai
import os
from dotenv import load_dotenv
import json
import logging
from datetime import datetime

class WhatsAppBot:
    def __init__(self):
        self.openai_client = openai.OpenAI(
            api_key=os.getenv('OPENAI_API_KEY')
        )
        self.setup_logging()
    
    def process_message(self, message, user_id):
        """Process incoming WhatsApp messages intelligently"""
        try:
            # Analyze message intent
            intent = self.analyze_intent(message)
            
            # Generate appropriate response
            response = self.generate_response(message, intent)
            
            # Log interaction
            self.log_interaction(user_id, message, response, intent)
            
            return response
            
        except Exception as e:
            logging.error(f"Error processing message: {e}")
            return "I apologize, but I'm experiencing technical difficulties."
    
    def analyze_intent(self, message):
        """Use AI to analyze user intent"""
        prompt = f"Analyze this message and classify intent: {message}"
        response = self.openai_client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=[{"role": "user", "content": prompt}]
        )
        return response.choices[0].message.content

# Flask app setup
app = Flask(__name__)
CORS(app)
bot = WhatsAppBot()

@app.route('/webhook', methods=['POST'])
def webhook():
    """Handle incoming WhatsApp webhooks"""
    data = request.get_json()
    
    if 'entry' in data:
        for entry in data['entry']:
            for change in entry['changes']:
                if change['value'].get('messages'):
                    for message in change['value']['messages']:
                        response = bot.process_message(
                            message['text']['body'],
                            message['from']
                        )
                        # Send response back to WhatsApp
                        send_whatsapp_message(message['from'], response)
    
    return jsonify({'status': 'success'})

if __name__ == '__main__':
    app.run(debug=True, host='0.0.0.0', port=5000)
